package consoCarbonne;

import java.text.DecimalFormat;

public class BiensetServices extends ConsoCarbone {
	private double montant;
	private static DecimalFormat df2 = new DecimalFormat("#.##");
	private final static int biensEtServices = 682;
	
	public BiensetServices(double montant) {
		super();
		this.montant = montant;
	}
	
	public double getMontant() {
		return montant;
	}
	
	public void setMontant(double montant) {
		this.montant = montant;
	}

	public int getBienetServices(){
		return biensEtServices;
	}


	@Override
	public void setImpact() {
		impact = montant/1750.0;
	}
	

	@Override
	public String toString() {
		return "L'empreinte carbone due Ã  la consommation d'un bien ou service d'un montant de " + montant + "  est de " + df2.format(impact) + "TCO2eq";
	}

}
